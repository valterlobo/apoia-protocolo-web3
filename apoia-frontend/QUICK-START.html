1️⃣ Setup Inicial

Pré-requisitos

  • MetaMask instalado (ou outro wallet Web3)
  • Navegador moderno (Chrome, Firefox, Edge, Brave)
  • Sepolia ETH para gas (testnet)

Instalação

  1. Clone o repositório
  2. Copie os endereços dos contratos para web3-integration.js
  3. Abra apoia-protocol-frontend.html no navegador

Configurar Endereços dos Contratos

3️⃣ Trabalhar com Campanhas

Criar uma Nova Campanha

const campaignManager = new CampaignManager(web3);

async function criarCampanha() {
    try {
        const receipt = await campaignManager.createCampaign(
            "Educação Descentralizada",     // title
            "Plataforma de educação online com blockchain",
            10,                              // softCap (ETH)
            50,                              // hardCap (ETH)
            Date.now() + 90 * 24 * 60 * 60 * 1000 // deadline (90 dias)
        );
        
        console.log('✅ Campanha criada:', receipt.hash);
        return receipt;
    } catch (error) {
        console.error('❌ Erro:', error.message);
    }
}

criarCampanha();

Contribuir para uma Campanha

async function contribuir() {
    try {
        // 1. Validar preço do oráculo
        const oracle = new OracleManager(web3);
        const priceData = await oracle.getLatestPrice();
        
        if (!priceData.isFresh) {
            console.warn('⚠️ Preço stale, aguarde atualização');
            return;
        }
        
        console.log('📊 Preço ETH/USD:', priceData.price);
        
        // 2. Contribuir
        const campaignAddress = '0x...'; // Endereço da campanha
        const amount = 0.5; // ETH
        
        const receipt = await campaignManager.contributeToCampaign(
            campaignAddress,
            amount
        );
        
        console.log('✅ Contribuição confirmada:', receipt.hash);
        console.log('🎁 Você receberá NFT em breve!');
        
        return receipt;
    } catch (error) {
        console.error('❌ Erro:', error.message);
    }
}

contribuir();

Buscar Detalhes da Campanha

async function buscarDetalhes(campaignAddress) {
    try {
        const details = await campaignManager.getCampaignDetails(campaignAddress);
        
        console.log('📋 Detalhes da Campanha:');
        console.log('Título:', details.title);
        console.log('Criador:', details.creator);
        console.log('Soft Cap:', details.softCap, 'ETH');
        console.log('Hard Cap:', details.hardCap, 'ETH');
        console.log('Arrecadado:', details.raised, 'ETH');
        
        const progress = (details.raised / details.hardCap * 100).toFixed(1);
        console.log('Progresso:', progress + '%');
        
    } catch (error) {
        console.error('❌ Erro:', error.message);
    }
}

Vesting de AGT

// Criar vesting (apenas minter)
async function criarVesting() {
    try {
        const receipt = await tokenManager.createVestingSchedule(
            '0xabcd...efgh',              // beneficiary
            100000,                        // amount AGT
            30 * 24 * 60 * 60,            // cliff (30 dias)
            365 * 24 * 60 * 60            // duration (1 ano linear)
        );
        
        console.log('✅ Vesting criado:', receipt.hash);
        console.log('Beneficiary poderá resgatar após 30 dias');
        
    } catch (error) {
        console.error('❌ Erro:', error.message);
    }
}

// Liberar vesting (beneficiary ou owner)
async function liberarVesting() {
    try {
        const receipt = await tokenManager.releaseVesting('0xabcd...efgh');
        
        console.log('✅ Vesting liberado:', receipt.hash);
        console.log('AGT disponível na wallet');
        
    } catch (error) {
        console.error('❌ Erro:', error.message);
    }
}

criarVesting();
// Após 30 dias...
liberarVesting();
💡 Dica: Sempre aprove spend antes de fazer stake. Use números exatos (sem decimais para AGT).

5️⃣ Fazer Stake de AGT

Flow Completo de Staking

const stakingManager = new StakingManager(web3);

async function stakeFlow() {
    try {
        // 1. Aprovar tokens
        console.log('📝 Etapa 1: Aprovando 50,000 AGT...');
        await tokenManager.approve(
            '0x4234567890123456789012345678901234567893', // Staking
            50000
        );
        console.log('✅ Aprovado');
        
        // 2. Fazer stake por 180 dias
        console.log('📝 Etapa 2: Fazendo stake...');
        const receipt = await stakingManager.stake(
            50000,                  // amount
            180 * 24 * 60 * 60      // duration (180 dias)
        );
        console.log('✅ Staked:', receipt.hash);
        
        // 3. Verificar informações
        console.log('📝 Etapa 3: Obtendo informações...');
        const stakedAmount = await stakingManager.getStakedAmount();
        console.log('Amount staked:', stakedAmount, 'AGT');
        
        const multiplier = await stakingManager.getVotingMultiplier();
        console.log('Voting multiplier:', multiplier, 'x');
        
        const rewards = await stakingManager.calculateReward();
        console.log('Rewards disponíveis:', rewards, 'AGT');
        
        return receipt;
    } catch (error) {
        console.error('❌ Erro:', error.message);
    }
}

stakeFlow();

Resgatar Rewards

async function resgatarRewards() {
    try {
        // Verificar rewards pendentes
        const rewards = await stakingManager.calculateReward();
        console.log('💰 Rewards disponíveis:', rewards, 'AGT');
        
        if (parseFloat(rewards) > 0) {
            // Resgatar
            const receipt = await stakingManager.claim();
            console.log('✅ Rewards resgatados:', receipt.hash);
        } else {
            console.log('⏳ Nenhum reward disponível ainda');
        }
        
    } catch (error) {
        console.error('❌ Erro:', error.message);
    }
}

resgatarRewards();

Desfazer Stake

async function desfarzeStake() {
    try {
        const receipt = await stakingManager.unstake();
        console.log('✅ Stake removido:', receipt.hash);
        console.log('AGT retornado à sua wallet');
        
    } catch (error) {
        console.error('❌ Erro:', error.message);
    }
}

desfarzeStake();
Duração Multiplicador Caso de Uso
30 dias 1.5x Curto prazo, mais flexível
90 dias 2.0x Médio prazo
180 dias 2.5x Longo prazo, máximo prêmio
365 dias 3.0x Máximo comprometimento

6️⃣ Votar na DAO

Votar em Proposta

const daoManager = new DAOManager(web3);

async function votarNaProposta() {
    try {
        // Verificar seu poder de voto
        const votingPower = await stakingManager.getVotingMultiplier();
        console.log('🗳️ Seu poder de voto: ' + votingPower + 'x');
        
        // Votar
        // 0 = Contra, 1 = A Favor, 2 = Abstenção
        const proposalId = 1;
        const support = 1; // A Favor
        
        const receipt = await daoManager.castVote(proposalId, support);
        console.log('✅ Voto confirmado:', receipt.hash);
        
        return receipt;
    } catch (error) {
        console.error('❌ Erro:', error.message);
    }
}

votarNaProposta();

Criar Proposta

async function criarProposta() {
    try {
        // Exemplo: Aumentar taxa para 6%
        const targets = ['0x1234...5678']; // Contrato alvo
        const values = [0]; // Sem ETH
        
        // Encoded function call: setFee(600)
        const calldatas = ['0x...'];
        
        const description = 'Aumentar taxa de plataforma de 5% para 6%';
        
        const receipt = await daoManager.propose(
            targets,
            values,
            calldatas,
            description
        );
        
        console.log('✅ Proposta criada:', receipt.hash);
        
        return receipt;
    } catch (error) {
        console.error('❌ Erro:', error.message);
    }
}

criarProposta();

Verificar Estado da Proposta

async function verificarProposta() {
    try {
        const proposalId = 1;
        
        // Estado
        const state = await daoManager.getProposalState(proposalId);
        console.log('Estado:', state);
        
        // Detalhes
        const details = await daoManager.getProposalDetails(proposalId);
        console.log('Descrição:', details.description);
        console.log('Votos a favor:', details.forVotes);
        console.log('Votos contra:', details.againstVotes);
        
        const total = parseFloat(details.forVotes) + parseFloat(details.againstVotes);
        const percentual = (parseFloat(details.forVotes) / total * 100).toFixed(1);
        console.log('Aprovação:', percentual + '%');
        
    } catch (error) {
        console.error('❌ Erro:', error.message);
    }
}

verificarProposta();
💡 Dica: Quanto maior seu stake, maior seu poder de voto. Combine isso com multiplicadores (1.5x - 3.0x) para máxima influência.

Logs Detalhados

// Ativar logs detalhados
function enableDebugLogs() {
    const originalLog = console.log;
    
    console.log = function(...args) {
        const timestamp = new Date().toLocaleTimeString();
        originalLog(`[${timestamp}]`, ...args);
    };
}

enableDebugLogs();

// Agora todos os logs incluirão timestamp
console.log('Ação realizada');
